## Map

class ComplexObject {
    // ðŸ§ 
}

class IDGenerator {
		constructor () {
				this.ids = new Map();
		}

		getID (complexObject) {
				if (this.ids.has(complexObject)) {
						return this.ids.get(complexObject);
                }

				const id = this.ids.size;
				this.ids.set(complexObject, id);

                return id;
		}
}
